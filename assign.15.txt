<?php
// Example number
$number = 12345;
$sum = 0;
$temp = $number;

// Loop until the number becomes 0
while ($temp > 0) {
    $digit = $temp % 10; // Get last digit
    $sum += $digit;      // Add digit to sum
    $temp = (int)($temp / 10); // Remove last digit
}

echo "The sum of digits of $number is: $sum";
?>
