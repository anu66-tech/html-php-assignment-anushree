<?php

$size = 3;
echo "   ";
for ($j = 1; $j <= $size; $j++) {
    echo str_pad($j, 6, " ", STR_PAD_LEFT);
}
echo "\n";
for ($i = 1; $i <= $size; $i++) {
    
    echo $i . " ";
    
    for ($j = 1; $j <= $size; $j++) {
        
        $result = round($i / $j, 2);
      
        echo str_pad($result, 6, " ", STR_PAD_LEFT);
    }
    echo "\n";
}
?>
