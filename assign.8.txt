<?php
function deal($priceA, $sizeA, $priceB, $sizeB) {
    // Calculate cost per liter
    $costPerLitreA = $priceA / $sizeA;
    $costPerLitreB = $priceB / $sizeB;

    echo "Drink A: Rs. $priceA for $sizeA L → Cost per litre: Rs. " . number_format($costPerLitreA, 2) . "<br>";
    echo "Drink B: Rs. $priceB for $sizeB L → Cost per litre: Rs. " . number_format($costPerLitreB, 2) . "<br><br>";

    // Compare and decide
    if ($costPerLitreA < $costPerLitreB) {
        echo "You should choose Drink A to save money!";
    } elseif ($costPerLitreB < $costPerLitreA) {
        echo "You should choose Drink B to save money!";
    } else {
        echo "Both drinks have the same value for money.";
    }
}

// Given values
$priceA = 25; $sizeA = 1.1; // 1.1 liters
$priceB = 23; $sizeB = 0.9; // 0.9 liters

// Call the function
deal($priceA, $sizeA, $priceB, $sizeB);
?>
