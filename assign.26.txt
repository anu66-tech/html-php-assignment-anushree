<?php
echo "<table border='1' cellpadding='20' cellspacing='0'>";

for ($row = 1; $row <= 8; $row++) {
    echo "<tr>";
    for ($col = 1; $col <= 8; $col++) {
        // Alternate colors
        $total = $row + $col;
        if ($total % 2 == 0) {
            echo "<td style='background-color: black;'></td>";
        } else {
            echo "<td style='background-color: white;'></td>";
        }
    }
    echo "</tr>";
}

echo "</table>";
?>
